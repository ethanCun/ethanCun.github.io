三、 打包部署到tomcat

				1. 移除格式校验： 
				webpack.base.conf.js -》 module -》 rules 注释掉Eslint
				
				2. config:
				dev.env.js:
				module.exports = merge(prodEnv, {
				  NODE_ENV: '"development"',
				  BASE_API: '"http://ocalhost:8080/"'  
				})
				
				prod.env.js:
				'use strict'
				module.exports = {
				  NODE_ENV: '"production"',
				  BASE_API: '"http://localhost:8080"' //线上路径，替换开发者路径
				}
				
				3. 设置统一的请求路径:
				build -》 utils -》 
				
				// Extract CSS when that option is specified
				// (which is the case during production build)
				if (options.extract) {
				  return ExtractTextPlugin.extract({
				    use: loaders,
				    fallback: 'vue-style-loader',
				    publicPath : '../../'  //新添加的内容，路径配置
				  })
				} else {
				  return ['vue-style-loader'].concat(loaders)
				}
				
				4. 
				build: {
				  // Template for index.html
				  index: path.resolve(__dirname, '../dist/index.html'),
				
				  // Paths
				  assetsRoot: path.resolve(__dirname, '../dist'),
				  assetsSubDirectory: 'static',
				  assetsPublicPath: './',
					
				  ...
				}
				
				5. 打包： npm run build
				
				6. 将dist下的文件放入到webapps下新建的vue文件夹下
				
				7. cd到tomcatbin目录， ./startup.sh启动项目， 访问http://localhost:8080/vue, 
				默认访问index.html
