2. File常用方法


         //文件
        File file = new File("/Users/macofethan/Desktop/javase/javase01/src/main/java/com/ethan/File文件与文件夹/a");

        //是否存在
        System.out.println(file.exists());

        if (!file.exists()){

            //以文件的形式创建
            file.createNewFile();

            //以目录的形式创建
            //file.mkdir();

            //以目录的形式创建, 可以建立不存在的子包
            //file.mkdirs();
        }


        //获取文件的父路径
        System.out.println(file.getParent());
        //获取父文件对象
        //file.getParentFile();
        //获取文件的绝对路径
        System.out.println(file.getAbsolutePath());
        //文件名
        System.out.println(file.getName());
        //是否是文件
        System.out.println(file.isFile());
        //是否是目录
        System.out.println(file.isDirectory());
        //文件最后修改时间 1970到现在的毫秒数
        Date date = new Date(file.lastModified());
        SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss SSS");
        System.out.println(simpleDateFormat.format(date));
        //文件总字节数
        System.out.println(file.length());
        //修改名称
        file.renameTo(new File("/Users/macofethan/Desktop/javase/javase01/src/main/java/com/ethan/File文件与文件夹/abc.txt"));

        //删除
        //file.delete();


        //获取文件夹下的所有子目录:file.listFiles()
        File file1 = new File("/Users/macofethan/Desktop/javase/javase01/src/main/java/com/ethan/File文件与文件夹");

        for (File subFile : file1.listFiles()) {

            System.out.println(subFile.getAbsolutePath());
        }
