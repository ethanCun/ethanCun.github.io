2. 装箱与拆箱


            Integer aa = new Integer(100); //装箱
            System.out.println(aa.floatValue()); //拆箱

            Integer b = 100; //jdk1.5之后支持自动装箱
            int c = b; //jdk1.5之后支持自动拆箱

            //分析
            Integer d = 1000;
            Integer e = 1000;
            //相当于
            //Integer e = new Integer(1000);
            System.out.println(d == e); //false

            int f = 100;
            int g = 100;
            System.out.println(f == g); //true

            /走了缓存 Java中为了提高程序运行效率，会将-128 -- +127提前创建好， 放到
            //IntegerCache.cache中, 这个保存在方法区的整数型常量池中，这个区间的不在堆上面
            //创建值了，直接用整数型常量池中的数据
            //注意： 可以在VM option中设置Cache.high=-Djava.lang.Integer.IntegerCache.high=128
            Integer h = Integer.valueOf(127);
            Integer j = Integer.valueOf(127);
            System.out.println(h == j); //true

            //超出范围 不走缓存
            Integer k = Integer.valueOf(128);
            Integer l = Integer.valueOf(128);
            System.out.println(k == l); //false

            //这两个相当于上面
            Integer q = 127;
            Integer w = 127;
            System.out.println(q == w); //true

            Integer t = 128;
            Integer r = 128;
            System.out.println(t == r); //false

            //引用堆上的同一个对象， 地址相同
            System.out.println(h == q); //true
